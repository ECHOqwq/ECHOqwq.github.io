typedef struct ARR
{
struct ARR front;
int data;
struct ARR tail;
}*arr,node;

arr CreatHead(void)
{
int len;
arr head=(arr)malloc(sizeof(node));
head->front=head->tail=head;
printf (“请输入链表的长度:”);
scanf (“%d”,&len);
while (len)
{
arr current=(arr)malloc(sizeof(node));
printf (“请输入数据:”);
scanf (“%d”,&current->data);
current->front=head;
current->tail=head->tail;
head->tail->front=current;
head->tail=current;
len–;
}
return head;
}

pt->front->tail=pt->tail;

}
