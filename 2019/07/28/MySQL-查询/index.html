Mysql-查询

发表于 2019-07-28 | 更新于 2019-07-29 | 分类于 mysql | 评论数：

  1. 基本查询
    <1> 查看所有列

    SELECT * FROM tab_name;

    <2> 查看指定列

    查询表tab_name表中的指定列col_name1,col_name2...
SELECT col_name1 , col_name2, ... FROM tab_name;

    <3> 去掉重复值

    #使用DISTINCT将指定列的重复值剔除，来保证重复的值只查询出现一次
SELECT DISTINCT col_name FROM tab_name;

    <4> 查询语句中使用算术表达式

    #在sql中使用算数表达式
SELECT math_operation FROM tab_name;
e.g 
    # 计算员工奖金增加500后的奖金结果
    SELECT ename , comm + 500 FROM emp;

    <5> 查询语句中使用列的别名

    #使用别名查询
SELECT col_name AS alias_name, col_name2 alias_name2... FROM tab_name;
e.g 
    # 计算员工奖金增加500后的奖金结果
    SELECT ename AS '姓名', comm + 500 '预期奖金' FROM emp;

  2. 限制查询
    <1> where子句中使用简单比较操作符

    #使用简单比较操作符
SELECT * FROM tab_name WHERE col_name >value;
e.g
    #查询公司中薪水大于2000的所有员工
    SELECT * FROM emp WHERE sal >2000;

    <2> where子句中使用BETWEEN…AND

    #使用BETWEEN...AND
SELECT * FROM tab_name WHERE col_name BETWEEN start_value AND end_value;
e.g
    # 查询公司中奖金在1000~5000之间的所有员工
    SELECT * FROM emp WHERE comm BETWEEN 1000 AND 5000;

    <3> where子句中使用IN

    # 查询tab_name表中col_name为value1,value2,value3...的记录
SELECT * FROMtab_name
WHERE col_name IN(value1, value2, value3 ...);
e.g
    # 查询公司中职位为项目经理，项目组长、项目助理和程序员的员工
    SELECT * FROM emp
        WHERE job IN('项目经理','项目组长','项目助理','程序员');

    <4> where子句中使用LIKE

    #模糊查询
SELECT * FROM  tab_name WHERE col_name LIKE ‘%key%’;
e.g
    # 查询公司中所有助理
    SELECT * FROM emp WHERE job LIKE '%助理';

    <5> where子句中使用IS NULL符

    #判断某列为空
SELECT * FROM tab_name WHERE col_name IS NULL;
e.g
    # 查询公司中姓名为空的不合法数据
    SELECT * FROM emp WHERE ename IS NULL;

    <6> where子句中使用AND

    # AND条件并列查询
SELECT * FROM tab_name WHERE condition1 AND condition2;
e.g 
    # 查询公司中入职时间大于10年并且工资低于6000的员工
    SELECT * FROM emp WHERE hirdate < date_add(now() , interval -10 year) 
        AND sal < 6000;

    <7> where子句中使用OR

    #使用OR进行多条件或查询
SELECT * FROM tab_name WHERE condition1 OR condition2;
e.g
    # 查询公司中姓名为NULL或者为空的员工
    SELECT * FROM emp WHERE ename IS NULL OR ename = '';

    <8> where子句中使用NOT

    # 使用NOT关键词
SELECT * FROM tab_name WHERE col_name IS NOT NULL;
e.g
    # 查询公司中所有工资和奖金不为NULL的员工
    SELECT * FROM emp WHERE sal IS NOT NULL AND comm IS NOT NULL;

    <9> 优先级

    比较运算符>NOT > AND > OR
  3. 查询结果排序
    <1> 降序排列
    # 按照指定列col_name的降序排列查询结果
SELECT * FROM tab_name ORDER BY col_name DESC;
e.g
    # 按照薪水的降序排列查询结果
    SELECT * FROM emp ORDER BY sal DESC;
    <2> 升序排列
    #按照指定列的升序排列查询结果
SELECT * FROM tab_name ORDER BY col_name ASC;
e.g
    # 公司要求按照奖金升序查看员工信息
    SELECT * FROM emp ORDER BY comm ASC;
    <3> 多列排序
    # 指定多列的排序查询
SELECT * FROM tab_name ORDER BY col_name1 ASC/DESC, col_name2 ASC/DESC, ...;
e.g
    # 按照公司员工的入职时间顺序排列，相同入职时间的按照工资降序排列
    SELECT * FROM emp ORDER BY hirdate ASC , sal DESC;
  4. 聚合函数查询
    <1> COUNT函数(用来计算查询的总数 )
    # 查询公司所有员工
SELECT COUNT(*) FROM emp;
    <2> AVG、SUM函数
    # 查询公司所有员工平均薪水和总薪水
SELECT AVG(sal), SUM(sal) FROM emp;
    <3> MAX、MIN函数
    # 查看公司中员工的最高薪水和最低薪水
SELECT MAX(sal), MIN(sal) FROM emp;
  5. 分组查询
    <1> GROUP…BY短语
    # GROUP BY 分组函数语法
SELECT<*, column [alias],...>
FROM tab_name
[WHERE conditions]
[GROUP BY group_expression]
[HAVING group_condition]
[ORDER BY column [ASC|DESC]]
e.g
    # 查询公司中每个部分的平均薪水
    SELECT deptno, AVG(sal) FROM emp GROUP BY deptno;
    <2> HAVING子句
    #查询公司中每个部门的平均工资大于4000的部门
SELECT deptno, AVG(sal) FROM emp
    GROUP BY deptno HAVING AVG(sal) > 4000;
  6. 连接查询
    <1> 笛卡尔积
    多行表在查询时，如果定义了无效连接或者漏写了连接条件，就会产生笛卡尔乘积现象
所谓笛卡尔乘积就是每个表的每一行都和其他表的每一行组合
假设两张表的总行数分别是X行和Y行，笛卡尔乘积就会返回X*Y行记录
    <2> e.g
    #笛卡尔乘积现象
注:emp表中有10条数据，dept表中有3条
   SELECT COUNT(*)FROM emp,dept;
   out:30
    <3> 小链接
    给表一个别名
e.g
    SELECT COUNT(*) FROM emp e,dept d WHERE e.deptno=d.deptno;
注:
    在连接查询时，使用[表的全名.字段名称]这样的表示方式显得臃肿也不方便
    可以给表定义别名，然后使用[别名.字段]来简化连接查询
    <4> 等值连接
    #使用等值连接查询两张表的数据
SELECT t1.col_name1, t2.col_name2
FROM table1t1 , table2 t2
WHERE  t1.col_name1 = t2.col_name2;
e.g
    # 使用等值连接查询销售部所有员工信息
    SELECT * FROM emp e, dept d WHERE e.deptno = d.deptno AND d.dname='销售部';
    <5> 自连接
    #自连接查询语法
SELECT * FROM tab_name a , tab_name b
WHERE a.col_name1 = b.col_name2, ...
e.g
    # 查询公司员工和所属上级的信息
    SELECT e1.empno,e1.ename,e2.empno,e2.ename
        FROM emp e1, emp e2 
        WHERE e1.mgr = e2.empNo;
    <6> 内连接和外连接
    #各种连接查询的语法结构
SELECT * FROM table1 t1
[INNER|LEFT|RIGHT]JOIN table2 t2
ON t1.col_name = t2.col_name;
注:
   前面提到的等值连接和自连接也是属于内连接范畴的查询方式
   语法中的INNER是属于内连接查询的，LEFT|RIGHT都属于外连接查询
区别
   内连接查询返回所有满足连接条件的记录
   外连接查询是内连接查询的扩展
   除了那些满足条件的数据，还可以返回不满足条件的的连接符指定的表的数据记录
内连接查询:
    SELECT d.deptno,d.dname,e.empno,e.ename FROM emp e
    INNER JOIN dept d
    ON e.deptno = d.deptno
    ORDER BY d.deptno;
    注:
       在内连接查询时，返回的结果都是符合连接条件的查询数据
       这样的前提下，查询到的数据就会不满足需求要求
       编号40的后勤部是没有员工的，但是不能因为没有员工就不列出后勤部这个部门
左外连接查询:
    # 查询公司员工和所属上级的信息
    SELECT d.deptno,d.dname,e.empno,e.ename FROM dept d
        LEFT JOIN emp e
        ON d.deptno = e.deptno
        ORDER BY d.deptno;
    注:
       左外连接查询，查询时除了返回满足连接条件的数据
       也返回LEFT指定的左边的表的不满足连接的数据
       如编号40的后勤部数据
右外连接查询:
    即把上面LEFT改为RIGHT
  7. 子查询
    <1> 引入
    WHERE查询条件中的限制条件不是一个确定的值，而是来自另外一个查询的结果
比如公司要查询比员工吴用工资高的员工
这样的话首先要获取吴用的工资，然后用这个工资来做条件查询其他的员工
    <2> 子查询
    为了给主查询语句提供数据而首先执行的查询语句称为子查询
    <3> 单行子查询
    # 查询软件部门下的所有员工信息
SELECT * FROM emp e
WHERE e.deptno = (SELECT d.deptno FROM dept d WHERE d.dname='软件部');
    <4> 多行子查询IN
    # 使用IN操作符——查询公司所有的员工都分布在哪些部门
SELECT * FROM dept d
    WHERE d.deptno IN (SELECT e.deptno FROMemp e);
    <5> 多行子查询ANY
    # 找出工资最高的人(比任意人都高)
SELECT * FROM emp e1
    WHERE e1.sal >ANY(SELECT e2.sal FROM emp e2);
    <6> 多行子查询ALL
    # 查询公司中比所有的助理工资高但不是助理的员工
SELECT * FROM emp e1
    WHERE e1.sal >ALL(SELECT e2.sal FROM emp e2 
            WHERE e2.job LIKE '%助理');
    <7> 多列子查询
    # 查询公司中和员工张青相同薪水和奖金的员工
SELECT * FROM emp e1 
    WHERE (e1.sal,e1.comm) = (SELECT e2.sal,e2.comm FROM emp e2 WHERE e2.ename='张青');
