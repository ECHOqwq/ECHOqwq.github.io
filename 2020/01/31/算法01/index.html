算法01——多数相加之和问题

发表于 2020-01-31 | 分类于 Leetcode | 评论数：

注：以下算法/思路参考相关题的题解

  1. 两数之和
    情境：你来到了一个相亲角，里面有一个主持人负责收集男女嘉宾的个人信息以及心仪对象的信息
    方法1：暴力法
    主持人从男嘉宾的资料开始翻起，每翻到一个男嘉宾就去找女嘉宾那一叠信息，直到找到这位男嘉宾心仪的女嘉宾才终止此次循环。
    public int[] twoSum(int[] nums, int target) {
    for (int i = 0; i < nums.length; i++) {
    for (int j = i + 1; j < nums.length; j++) {
        if (nums[j]+nums[i]==target) {
            return new int[] {i,j};
        }
    }
}
throw new IllegalArgumentException("error");
    }
    方法2：两遍哈希表
    主持人先把男女嘉宾的个人信息和他们心仪对象的信息录入一个档案里，然后从头开始看这个档案，每找一个人就可以从档案中获取ta心仪对象的信息，直接配对。
    public int[] twoSum(int[] nums, int target) {
    Map<Integer, Integer> map = new HashMap<>();
for (int i = 0; i < nums.length; i++) {
    map.put(nums[i], i);
}
for (int i = 0; i < nums.length; i++) {
    int complement = target - nums[i];
    if (map.containsKey(complement) && map.get(complement) != i) {
        return new int[] {i,map.get(complement)};
    }
}
throw new IllegalArgumentException("error");
    }
    方法3：一遍哈希表
    主持人每喊到一个嘉宾名字，嘉宾就上去告诉主持人ta的个人信息和心仪对象，主持人从档案里看，如果该嘉宾已经是录入过档案的人的心仪对象，则直接配对成功，若不是则把该嘉宾的信息录入档案。
    public int[] twoSum(int[] nums, int target) {
    Map<Integer, Integer> map = new HashMap<>();
for (int i = 0; i < nums.length; i++) {
    int complement = target - nums[i];
    if (map.containsKey(complement)) {
        return new int[] {map.get(complement),i};
    }
    map.put(nums[i], i);
}
throw new IllegalArgumentException("error");
    }
  2. 三数之和
    方法1：暴力法
    同两数之和一样，只是多加了一层循环
    方法2：双指针联动
    大致思路：
    i指针和nums[]数组长度限定大区间，L和R跑小区间
public List<List<Integer>> threeSum(int[] nums) {
 List<List<Integer>> list = new ArrayList<>();
 if (nums==null||nums.length<3){
     return list;
 }
//排序
Arrays.sort(nums);
int len = nums.length;
if((nums[0]+nums[1]+nums[2]>0)||(nums[len-1]+nums[len-2]+nums[len-3]<0)){
    return list;
}
//双指针
for(int i=0;i<len-2;i++) {
    if(nums[i] > 0){
        return list;
    }
    //跳过重复值
    if(i > 0 && nums[i] == nums[i-1]){
        continue;
    }
    int L = i+1, R = len-1;
    while (L<R) {
        int tmp = nums[i] + nums[L] + nums[R];
        if(tmp==0) {
            list.add(Arrays.asList(nums[i],nums[L],nums[R]));
            while(L<R&&nums[L+1]==nums[L]){
                L++;
            }
            while(L<R&&nums[R-1]==nums[R]){
                R--;
            }
            L++;
            R--;
        } else if(tmp<0) {
            L++;
        } else {
            R--;
        }
    }
}
return list;
    }
  3. 四数之和
    注意：一定要分清continue和break的用法！！！
    大致思路：
    大前提：数组排序
四指针联动跑，其中a、b两根指针负责高频率地在j和nums[]数组长度确定的区间内跑动，i指针控制j指针的走向
i：0——length-3
j: i+1——length-2
a：j+1
b:length-1
若cur<target，a指针向后移一个
若cur>target，b指针向前移一个
若cur==target，ab指针同时移动
在移动四个指针时需要注意，若指针移到的下一位上的值和前一位相同，则跳过该位，继续寻找下一个(避免出现值重复的情况)
    //判断最小的四个数加起来是否比目标值大，若大则返回true
    public static boolean JudgeMin(int[] nums,int target,int a,int b,int c,int d) {
    if (nums[a]+nums[b]+nums[c]+nums[d]>target) {
    return true;
}
return false;
    }
    //判断最大的四个数加起来是否比目标值小，若小则返回true
    public static boolean JudgeMax(int[] nums,int target,int a,int b,int c,int d) {
    if (nums[a]+nums[b]+nums[c]+nums[d]<target) {
    return true;
}
return false;
    }
    public static List<List> fourSum(int[] nums, int target) {
    List<List<Integer>> list = new ArrayList<>();
//如果nums数组没有元素或者少于四个，则直接退出
if(nums==null||nums.length<4){
    return list;
}
//排序以便通过最小值最大值缩短所需时间
Arrays.sort(nums);
int length = nums.length;
//如果排序后最大的四个数相加小于目标值，则直接返回
if (JudgeMax(nums,target,length-1,length-2,length-3,length-4)) {
    System.out.println("不存在");
    return list;
}
int a,b; //a，b用于跑动
for (int i = 0;i<length-3;i++) { //最外的指针
    //用于跳过重复值
    if (i>0&&nums[i]==nums[i-1]) {
        continue;
    }
    if ((JudgeMin(nums,target,i,i+1,i+2,i+3))||(JudgeMax(nums,target,i,length-1,length-2,length-3))) {
        continue;
    }
    for (int j = i+1;j<length-2;j++) {
        //如果该元素和它的前一个相等，则跳过，避免重复
        if (j>i+1&&nums[j]==nums[j-1]) {
            continue;
        }
        a = j+1;
        b = length-1;
        if((JudgeMin(nums,target,i,j,a,a+1))||(JudgeMax(nums,target,i,j,b,b-1))) {
            continue;
        }
        while (a<b) {
            int cur = nums[i]+nums[j]+nums[a]+nums[b];
            if (cur==target) {
                list.add(Arrays.asList(nums[i],nums[j],nums[a],nums[b]));
                a++;
                while (a<b&&nums[a]==nums[a-1]) {
                    a++;
                }
                b--;
                while(a<b&&nums[b]==nums[b+1]) {
                    b--;
                }
            }else if(cur>target) {
                b--;
            }else {
                a++;
            }
        }
    }
}
return list;
    }
