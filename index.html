typedef struct ARR
{
struct ARR front;
int data;
struct ARR tail;
}*arr,node;

arr CreatHead(void)
{
int len;
arr head=(arr)malloc(sizeof(node));
head->front=head->tail=head;
printf (“请输入链表的长度:”);
scanf (“%d”,&len);
while (len)
{
arr current=(arr)malloc(sizeof(node));
printf (“请输入数据:”);
scanf (“%d”,&current->data);
current->front=head;
current->tail=head->tail;
head->tail->front=current;
head->tail=current;
len–;
}
return head;
}

arr CreatTail(void)
{

head->front=NULL;

void Insert(arr head)
{
int data;
arr current,pt;

void Delete(arr head)
{
int data;
printf (“请输入你要删除的数据:”);
scanf (“%d”,&data);
arr pt=head->tail;
while (pt!=head)
{
if (pt->data==data)
{
pt->front->tail=pt->tail;
pt->tail->front=pt->front;
return;
}
pt=pt->tail;
}
printf (“不存在该数据!\n”);
}

while (pt!=head)
